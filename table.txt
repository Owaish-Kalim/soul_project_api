CREATE TABLE slh_teams("TeamId" serial PRIMARY KEY, "Email" VARCHAR(50) UNIQUE NOT NULL, "FirstName" VARCHAR(50) NOT NULL, "LastName" VARCHAR(50) NOT NULL, "MobileNo" VARCHAR(10) UNIQUE NOT NULL, "Address" VARCHAR(50) NOT NULL, "CreatedAt" TIMESTAMP NOT NULL, "JoiningDate" TIMESTAMP WITHOUT TIME ZONE, "Status" VARCHAR(50) NOT NULL, "Password" TEXT NOT NULL, "Token" TEXT);
